import RPi.GPIO as GPIO
from mfrc522 import SimpleMFRC522
import time

reader = SimpleMFRC522()
read_attempts = []
BLOCK_THRESHOLD = 5  # Maximum allowed reads within 10 seconds
BLOCK_TIME = 10  # Time in seconds to block access

def is_attack_detected():
    now = time.time()
    read_attempts.append(now)
    read_attempts[:] = [t for t in read_attempts if now - t < 10]
    return len(read_attempts) > BLOCK_THRESHOLD

try:
    print("IDS Active... Monitoring RFID activity.")
    while True:
        id, text = reader.read()
        if is_attack_detected():
            print("‚ö†Ô∏è Intrusion detected! Blocking further reads for 10 seconds.")
            time.sleep(BLOCK_TIME)
            read_attempts.clear()
        else:
            print(f"Read ID: {id}, Data: {text}")

    üõ°Ô∏è 2. Intrusion Detection System (Detect Suspicious Activity)

    üìå What You‚Äôll Do: Block RFID access if an attack is detected (too many scans too fast).

    Step-by-Step:

    1Ô∏è‚É£ Create & Run the IDS Script

    python
CopyEdit
import RPi.GPIO as GPIO
from mfrc522 import SimpleMFRC522
import time

reader = SimpleMFRC522()
read_attempts = []
BLOCK_THRESHOLD = 5  # Max allowed reads in 10 seconds
BLOCK_TIME = 10  # Block time in seconds

def is_attack_detected():
    now = time.time()
    read_attempts.append(now)
    read_attempts[:] = [t for t in read_attempts if now - t < 10]
    return len(read_attempts) > BLOCK_THRESHOLD

try:
    print("IDS Active... Monitoring RFID activity.")
    while True:
        id, text = reader.read()
        if is_attack_detected():
            print("‚ö†Ô∏è Intrusion detected! Blocking further reads for 10 seconds.")
            time.sleep(BLOCK_TIME)
            read_attempts.clear()
        else:
            print(f"Read ID: {id}, Data: {text}")

finally:
    GPIO.cleanup()
